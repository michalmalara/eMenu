12@register.simple_tag(takes_context=True) 

13def param_replace(context, **kwargs): 

14 """ 

15 Function returns GET parameters to use with pagination next/prev data. 

16 It allows to keep search and ordering settings with pagination. 

17 """ 

18 d = context['request'].GET.copy() 

19 for k, v in kwargs.items(): 

20 d[k] = v 

21 return d.urlencode()
